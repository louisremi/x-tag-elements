Here's some CSS

p { color: red; }
.alt { color: blue; }

Here's some JS

// toggle the class 'alt' on click
document.addEventListener('click', function( event ) {
    if ( event.target.nodeName == 'P' ) {
        event.target.classList.toggle('alt');
    }
});

And then some HTML

<p>Click me to toggle my color!</p>
